初始化

执行下面代码：

    //初始化模型组件
    new DeviceSelection({
        data:data,//构件数据列表、必须的
        height: "580px",//模型高度
        width:"1200px",//模型宽度
        projectCode: '20000305',//项目ProjectCode
        token: "",
        appKey: "",
        callback: function (res) {
            //回调函数
            //用户点击确认时、执行的回调、传回数据
            alert(JSON.stringify(res))
        }
    })

数据字段说明
